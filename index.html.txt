<script>
(function() {
  'use strict';

  const STORAGE_KEY = 'tradeConfluenceMeter_v1';

  const defaultData = {
    threshold: 70,
    sections: [
      {
        title: 'WEEKLY / DAILY',
        items: [
          { label: 'At AOI / Rejected', weight: 10, checked: false },
          { label: 'Touching EMA', weight: 5, checked: false },
          { label: 'Candlestick rejection', weight: 10, checked: false },
          { label: 'Rejection from previous structure point', weight: 10, checked: false },
          { label: 'Round number / psych level', weight: 5, checked: false },
          { label: 'Patterns', weight: 10, checked: false },
        ]
      },
      {
        title: '4HR CHECKLIST',
        items: [
          { label: 'Touching EMA', weight: 5, checked: false },
          { label: 'Candlestick rejection', weight: 10, checked: false },
          { label: 'Rejection from previous structure point', weight: 10, checked: false },
          { label: 'Patterns', weight: 10, checked: false },
        ]
      },
      {
        title: 'DAILY / 4HR',
        items: [
          { label: 'At AOI / Rejected', weight: 15, checked: false },
          { label: 'Touching EMA', weight: 5, checked: false },
          { label: 'Candlestick rejection', weight: 10, checked: false },
          { label: 'Rejection from previous structure point', weight: 15, checked: false },
          { label: 'Round number / psych level', weight: 5, checked: false },
          { label: 'Patterns', weight: 10, checked: false },
        ]
      }
    ]
  };

  let state = loadState();

  function loadState() {
    try {
      const raw = localStorage.getItem(STORAGE_KEY);
      if (!raw) return structuredClone(defaultData);
      const saved = JSON.parse(raw);
      // Merge missing sections from defaults
      const merged = structuredClone(saved);
      if (!merged.sections) merged.sections = [];
      const defaultSections = structuredClone(defaultData.sections);
      for (let i = 0; i < defaultSections.length; i++) {
        if (!merged.sections[i]) {
          merged.sections[i] = defaultSections[i];
        }
      }
      if (typeof merged.threshold !== 'number') merged.threshold = defaultData.threshold;
      return merged;
    } catch (e) {
      return structuredClone(defaultData);
    }
  }

  function saveState() {
    try {
      localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
    } catch (e) {}
  }

  function getSectionMax(section) {
    return section.items.reduce((s, it) => s + (Number(it.weight) || 0), 0);
  }

  function getSectionAchieved(section) {
    return section.items.reduce((s, it) => s + (it.checked ? (Number(it.weight) || 0) : 0), 0);
  }

  function getTotalMax() {
    return state.sections.reduce((s, sec) => s + getSectionMax(sec), 0);
  }

  function getTotalAchieved() {
    return state.sections.reduce((s, sec) => s + getSectionAchieved(sec), 0);
  }

  function getScore() {
    const max = getTotalMax();
    if (max === 0) return 0;
    return Math.round((getTotalAchieved() / max) * 100);
  }

  function getVerdict(score, threshold) {
    if (score >= threshold) {
      return { text: 'HIGH-PROBABILITY SETUP', cls: 'green' };
    }
    if (score >= threshold - 20) {
      return { text: 'MARGINAL — WAIT FOR MORE CONFLUENCE', cls: 'amber' };
    }
    return { text: 'LOW CONFLUENCE — SKIP', cls: 'red' };
  }

  // DOM refs
  const scoreNum = document.getElementById('scoreNum');
  const verdictPill = document.getElementById('verdictPill');
  const barFill = document.getElementById('barFill');
  const thresholdMarker = document.getElementById('thresholdMarker');
  const thresholdInput = document.getElementById('thresholdInput');
  const panelsContainer = document.getElementById('panels');

  // Render panels once
  function renderPanels() {
    panelsContainer.innerHTML = '';
    state.sections.forEach((section, secIdx) => {
      const panel = document.createElement('div');
      panel.className = 'panel';
      panel.dataset.sec = secIdx;

      const header = document.createElement('div');
      header.className = 'panel-header';
      header.innerHTML = `
        <div class="panel-title">${escapeHtml(section.title)}</div>
        <div class="panel-subtotal">
          <span class="sub-num" id="sub-${secIdx}">0</span> / <span id="max-${secIdx}">0</span>
        </div>
      `;
      panel.appendChild(header);

      const body = document.createElement('div');
      body.className = 'panel-body';
      body.id = `body-${secIdx}`;
      panel.appendChild(body);

      const footer = document.createElement('div');
      footer.className = 'panel-footer';

      const addBtn = document.createElement('button');
      addBtn.className = 'btn btn-add';
      addBtn.textContent = '+ Add item';
      addBtn.addEventListener('click', () => addItem(secIdx));
      footer.appendChild(addBtn);

      const resetBtn = document.createElement('button');
      resetBtn.className = 'btn btn-reset';
      resetBtn.textContent = 'Reset defaults';
      resetBtn.addEventListener('click', () => {
        if (confirm('Reset this section to default checklist?')) {
          resetSection(secIdx);
        }
      });
      footer.appendChild(resetBtn);

      panel.appendChild(footer);
      panelsContainer.appendChild(panel);

      renderSectionRows(secIdx);
    });
  }

  function renderSectionRows(secIdx) {
    const body = document.getElementById(`body-${secIdx}`);
    if (!body) return;
    body.innerHTML = '';
    const section = state.sections[secIdx];
    section.items.forEach((item, itemIdx) => {
      const row = document.createElement('div');
      row.className = 'row' + (item.checked ? ' checked' : '');
      row.dataset.item = itemIdx;

      const cbWrap = document.createElement('div');
      cbWrap.className = 'cb-wrap';
      const cb = document.createElement('input');
      cb.type = 'checkbox';
      cb.className = 'cb';
      cb.checked = item.checked;
      cb.addEventListener('change', () => {
        item.checked = cb.checked;
        row.classList.toggle('checked', item.checked);
        updateAll();
      });
      cbWrap.appendChild(cb);
      row.appendChild(cbWrap);

      const label = document.createElement('div');
      label.className = 'row-label';
      label.contentEditable = true;
      label.textContent = item.label;
      label.addEventListener('blur', () => {
        item.label = label.textContent.trim() || item.label;
        saveState();
      });
      label.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') { e.preventDefault(); label.blur(); }
      });
      row.appendChild(label);

      const wBox = document.createElement('div');
      wBox.className = 'weight-box';
      const wIn = document.createElement('input');
      wIn.type = 'number';
      wIn.className = 'weight-input';
      wIn.value = item.weight;
      wIn.min = 0;
      wIn.max = 100;
      wIn.step = 1;
      wIn.addEventListener('input', () => {
        item.weight = Number(wIn.value) || 0;
        updateAll();
      });
      wBox.appendChild(wIn);
      const pct = document.createTextNode('%');
      wBox.appendChild(pct);
      row.appendChild(wBox);

      const rm = document.createElement('button');
      rm.className = 'remove-btn';
      rm.innerHTML = '&times;';
      rm.title = 'Remove item';
      rm.addEventListener('click', () => {
        section.items.splice(itemIdx, 1);
        renderSectionRows(secIdx);
        updateAll();
      });
      row.appendChild(rm);

      body.appendChild(row);
    });
  }

  function addItem(secIdx) {
    const section = state.sections[secIdx];
    section.items.push({ label: '', weight: 5, checked: false });
    renderSectionRows(secIdx);
    updateAll();
    // Focus new label
    const body = document.getElementById(`body-${secIdx}`);
    const rows = body.querySelectorAll('.row');
    const lastRow = rows[rows.length - 1];
    if (lastRow) {
      const lbl = lastRow.querySelector('.row-label');
      if (lbl) { lbl.focus(); }
    }
  }

  function resetSection(secIdx) {
    const def = structuredClone(defaultData.sections[secIdx]);
    if (!def) return;
    state.sections[secIdx] = def;
    renderSectionRows(secIdx);
    updateAll();
  }

  function updateAll() {
    const score = getScore();
    const threshold = Number(thresholdInput.value) || 0;
    const verdict = getVerdict(score, threshold);

    scoreNum.textContent = score + '%';
    scoreNum.className = 'score-number ' + verdict.cls;

    verdictPill.textContent = verdict.text;
    verdictPill.className = 'verdict-pill ' + verdict.cls;

    barFill.style.width = score + '%';
    thresholdMarker.style.left = Math.min(100, Math.max(0, threshold)) + '%';

    // Update subtotals
    state.sections.forEach((sec, i) => {
      const subEl = document.getElementById(`sub-${i}`);
      const maxEl = document.getElementById(`max-${i}`);
      if (subEl) subEl.textContent = getSectionAchieved(sec);
      if (maxEl) maxEl.textContent = getSectionMax(sec);
    });

    saveState();
  }

  function escapeHtml(str) {
    const div = document.createElement('div');
    div.textContent = str;
    return div.innerHTML;
  }

  // Global reset
  function addGlobalReset() {
    const footer = document.querySelector('.app-footer');
    const link = document.createElement('a');
    link.href = '#';
    link.textContent = 'Reset all to defaults';
    link.style.cssText = 'color:var(--text-dim);margin-left:12px;text-decoration:underline;';
    link.addEventListener('click', (e) => {
      e.preventDefault();
      if (confirm('Reset the entire checklist to defaults? All custom edits will be lost.')) {
        state = structuredClone(defaultData);
        thresholdInput.value = state.threshold;
        renderPanels();
        updateAll();
      }
    });
    footer.appendChild(document.createTextNode(' · '));
    footer.appendChild(link);
  }

  // Init
  thresholdInput.value = state.threshold;
  thresholdInput.addEventListener('input', () => {
    state.threshold = Number(thresholdInput.value) || 0;
    updateAll();
  });

  renderPanels();
  updateAll();
  addGlobalReset();

})();
</script>

</body>
